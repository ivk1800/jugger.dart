import 'package:tests/inject/constructor/injected_constructor_with_positional_params.dart'
    as _i1;
import 'package:jugger/jugger.dart' as _i2;

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent.create();

  late final _i2.IProvider<double> _doubleProvider =
      _i2.Provider<double>(() => _i1.AppModule.provideNumberDouble());

  late final _i2.IProvider<int> _intProvider =
      _i2.Provider<int>(() => _i1.AppModule.provideNumberInt());

  late final _i2.IProvider<_i1.MyClass> _myClassProvider =
      _i2.Provider<_i1.MyClass>(
          () => _i1.MyClass(_intProvider.get(), _doubleProvider.get()));

  late final _i2.IProvider<String> _stringProvider = _i2.Provider<String>(
      () => _i1.AppModule.provideHello(_myClassProvider.get()));

  @override
  String get hello => _stringProvider.get();
