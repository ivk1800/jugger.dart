import 'package:tests/non_lazy/non_lazy_simple.dart' as _i1;
import 'package:jugger/jugger.dart' as _i2;

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent.create() {
    _initNonLazy();
  }

  late final _i2.IProvider<_i1.LazyRepository> _lazyRepositoryProvider =
      _i2.SingletonProvider<_i1.LazyRepository>(
          () => _i1.AppModule.provideLazyRepository());

  late final _i2.IProvider<_i1.NonLazyRepository> _nonLazyRepositoryProvider =
      _i2.SingletonProvider<_i1.NonLazyRepository>(
          () => _i1.AppModule.provideNonLazyRepository());

  @override
  _i1.LazyRepository getLazyRepository() => _lazyRepositoryProvider.get();
  @override
  _i1.NonLazyRepository getNonLazyRepository() =>
      _nonLazyRepositoryProvider.get();
  void _initNonLazy() {
    _nonLazyRepositoryProvider.get();
  }
}
