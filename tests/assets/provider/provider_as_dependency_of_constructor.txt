import 'package:tests/provider/provider_as_dependency_of_constructor.dart'
    as _i1;
import 'package:jugger/jugger.dart' as _i2;

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent.create();

  late final _i2.IProvider<_i1.Config> _configProvider =
      _i2.Provider<_i1.Config>(() => _i1.Config(_intProvider));

  late final _i2.IProvider<int> _intProvider =
      _i2.Provider<int>(() => _i1.AppModule.provideVersion());

  late final _i2.IProvider<String> _stringProvider = _i2.Provider<String>(
      () => _i1.AppModule.provideName(_configProvider.get()));

  @override
  String getName() => _stringProvider.get();
  @override
  int getVersion() => _intProvider.get();
}
