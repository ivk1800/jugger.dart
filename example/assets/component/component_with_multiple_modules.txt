import 'package:example/component/component_with_multiple_modules.dart' as _i1;
import 'package:jugger/jugger.dart' as _i2;

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent.create();

  late final _i2.IProvider<int> _intProvider =
      _i2.Provider<int>(() => _i1.Module2.provideInt());

  late final _i2.IProvider<String> _stringProvider =
      _i2.Provider<String>(() => _i1.Module1.provideString());

  @override
  int getInt() => _intProvider.get();
  @override
  String getString() => _stringProvider.get();
}
