import 'package:example/component/component_with_build_instance_dependency.dart'
    as _i1;

class JuggerAppComponentBuilder implements _i1.MyComponentBuilder {
  String? _string;

  @override
  _i1.MyComponentBuilder appComponent(String s) {
    _string = s;
    return this;
  }

  @override
  _i1.AppComponent build() {
    assert(_string != null);
    return JuggerAppComponent._create(_string!);
  }
}

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent._create(this._string);

  final String _string;

  @override
  String get string => _string;
}
