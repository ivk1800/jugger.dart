import 'package:example/bind/bind_from_another_module.dart' as _i1;
import 'package:example/bind/bind_from_module.dart' as _i2;
import 'package:example/non_lazy/non_lazy_without_non_lazy.dart' as _i3;
import 'package:jugger/jugger.dart' as _i4;

class JuggerAppComponent implements _i3.AppComponent {
  JuggerAppComponent.create() {
    _init();
  }

  late _i4.IProvider<_i3.LazyRepository> _lazyRepositoryProvider;

  @override
  _i3.LazyRepository getLazyRepository() {
    return _lazyRepositoryProvider.get();
  }

  void _init() {
    _initProvides();
  }

  void _initProvides() {
    _lazyRepositoryProvider = _i4.SingletonProvider<_i3.LazyRepository>(() {
      return _i3.AppModule.provideLazyRepository();
    });
  }

  @override
  void dispose() {}
}
