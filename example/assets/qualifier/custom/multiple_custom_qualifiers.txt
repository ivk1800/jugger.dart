import 'package:example/qualifier/custom/multiple_custom_qualifiers.dart'
    as _i1;
import 'package:jugger/jugger.dart' as _i2;

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent.create() {
    _init();
  }

  late _i2.IProvider<_i1.AppConfig> _qDevAppConfigProvider;

  late _i2.IProvider<_i1.AppConfig> _qProdAppConfigProvider;

  @override
  _i1.AppConfig getDevAppConfig() {
    return _qDevAppConfigProvider.get();
  }

  @override
  _i1.AppConfig getProdAppConfig() {
    return _qProdAppConfigProvider.get();
  }

  @override
  _i1.AppConfig get devAppConfig => _qDevAppConfigProvider.get();
  @override
  _i1.AppConfig get prodAppConfig => _qProdAppConfigProvider.get();
  void _init() {
    _initProvides();
  }

  void _initProvides() {
    _qDevAppConfigProvider = _i2.SingletonProvider<_i1.AppConfig>(() {
      return _i1.AppModule.provideDevAppConfig();
    });
    _qProdAppConfigProvider = _i2.SingletonProvider<_i1.AppConfig>(() {
      return _i1.AppModule.provideProdAppConfig();
    });
  }
}
