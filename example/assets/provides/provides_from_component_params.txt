import 'package:example/provides/provides_from_component_params.dart' as _i1;

class JuggerAppComponentBuilder implements _i1.AppComponentBuilder {
  _i1.MainRouter? _mainRouter;

  @override
  _i1.AppComponentBuilder createNewChatComponent(_i1.MainRouter router) {
    _mainRouter = router;
    return this;
  }

  @override
  _i1.AppComponent build() {
    assert(_mainRouter != null);
    return JuggerAppComponent._create(_mainRouter!);
  }
}

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent._create(this._mainRouter) {
    _init();
  }

  final _i1.MainRouter _mainRouter;

  @override
  _i1.MainRouter getMainRouter() {
    return _mainRouter;
  }

  void _init() {
    _initProvides();
  }

  void _initProvides() {}
}
