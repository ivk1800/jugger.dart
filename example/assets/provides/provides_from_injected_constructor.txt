import 'package:example/provides/provides_from_injected_constructor.dart'
    as _i1;
import 'package:jugger/jugger.dart' as _i2;

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent.create();

  late final _i2.IProvider<_i1.MainRouter> _mainRouterProvider =
      _i2.Provider<_i1.MainRouter>(() => _i1.MainRouter());

  late final _i2.IProvider<_i1.MainViewModel> _mainViewModelProvider =
      _i2.SingletonProvider<_i1.MainViewModel>(
          () => _i1.AppModule.provideMainViewModel(_mainRouterProvider.get()));

  @override
  _i1.MainViewModel getMainViewModel() => _mainViewModelProvider.get();
}
