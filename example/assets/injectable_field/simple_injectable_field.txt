import 'package:example/injectable_field/simple_injectable_field.dart' as _i1;
import 'package:jugger/jugger.dart' as _i2;

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent.create();

  late final _i2.IProvider<String> _stringProvider =
      _i2.SingletonProvider<String>(() {
    return _i1.AppModule.provideString();
  });

  @override
  void inject(_i1.InjectableClass c) {
    c.helloString = _stringProvider.get();
  }
}
