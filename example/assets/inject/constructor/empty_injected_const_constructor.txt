import 'package:example/inject/constructor/empty_injected_const_constructor.dart'
    as _i1;
import 'package:jugger/jugger.dart' as _i2;

class JuggerAppComponent implements _i1.AppComponent {
  JuggerAppComponent.create();

  late final _i2.IProvider<_i1.MyClass> _myClassProvider =
      _i2.Provider<_i1.MyClass>(() {
    return const _i1.MyClass();
  });

  late final _i2.IProvider<String> _stringProvider = _i2.Provider<String>(() {
    return _i1.AppModule.provideHello(_myClassProvider.get());
  });

  @override
  String get hello => _stringProvider.get();
}
